# Based on the algorithm presented at https://www.geeksforgeeks.org/how-to-encrypt-and-decrypt-strings-in-python/
import rsa

def encrypt_decrypt(the_string):

   public_key, private_key = rsa.newkeys(512)

   encrypted_string = rsa.encrypt(the_string.encode(), public_key)

   print("original string: ", the_string)
   print("encrypted string: ", encrypted_string)

   decrypted_string = rsa.decrypt(encrypted_string, private_key).decode()

   print("decrypted string: ", decrypted_string)

if __name__ == '__main__':
   the_string = str(input("Enter the string: "))
   encrypt_decrypt(the_string)
